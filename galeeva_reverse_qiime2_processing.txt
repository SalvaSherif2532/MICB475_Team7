## Primer trimming ##
qiime cutadapt trim-single \
            --i-demultiplexed-sequences galeeva_reverse_demux_seqs.qza \
            --p-cores 16 \
            --p-front GACTACHVGGGTWTCTAAT \
            --p-discard-untrimmed \
            --p-no-indels \
            --o-trimmed-sequences galeeva_reverse_reads_trimmed_pre.qza \
                        --verbose

## Adapter trimming ##
qiime cutadapt trim-single \
  --i-demultiplexed-sequences galeeva_reverse_reads_trimmed_pre.qza \
  --p-cores 16 \
  --p-adapter ATTAGAWACCCBDGTAGTCC \
  --p-no-indels \
  --o-trimmed-sequences galeeva_reverse_reads_trimmed.qza \
  --verbose

## Make into human readable file ##
qiime demux summarize \
            --i-data galeeva_reverse_reads_trimmed.qza \
            --o-visualization galeeva_reverse_reads_trimmed_summary.qzv

## DADA2 single-end denoising ##
qiime dada2 denoise-single \
    --i-demultiplexed-seqs galeeva_reverse_reads_trimmed.qza \
    --p-trim-left 0 \
    --p-trunc-len 231 \
    --p-n-threads 16 \
    --o-table dada2/galeeva_reverse_table.qza \
    --o-representative-sequences dada2/galeeva_reverse_rep-seqs.qza \
    --o-denoising-stats dada2/galeeva_reverse_denoising-stats.qza \
    --verbose

## Summarize ASV table to visualize sequence counts per sample ##
qiime feature-table summarize \
    --i-table dada2/galeeva_reverse_table.qza \
    --o-visualization dada2/galeeva_reverse_table_summary.qzv

## Summarize representative sequences ##
qiime feature-table tabulate-seqs \
    --i-data dada2/galeeva_reverse_rep-seqs.qza \
    --o-visualization dada2/galeeva_reverse_rep-seqs_summary.qzv

## Visualize DADA2 denoising stats ##
qiime metadata tabulate \
    --m-input-file dada2/galeeva_reverse_denoising-stats.qza \
    --o-visualization dada2/galeeva_reverse_denoising-stats_summary.qzv

## Make taxonomic_analysis folder in the data folder ##
if [ ! -d taxonomic_analysis ]; then
    mkdir -p taxonomic_analysis
fi

## Use the trained classifier (in the almomani processing code) to assign taxonomy to your reads (galeeva_reverse_rep-seqs.qza) ##
qiime feature-classifier classify-sklearn \
  --i-classifier taxonomic_analysis/classifier.qza \
  --i-reads galeeva_covid/dada2/galeeva_reverse_rep-seqs.qza \
  --o-classification taxonomic_analysis/galeeva_reverse_taxonomy.qza

## Make into human readable file ##
qiime metadata tabulate \
  --m-input-file galeeva_reverse_taxonomy.qza \
  --o-visualization galeeva_reverse_taxonomy.qzv

## Taxonomy barplots ##
qiime taxa barplot \
  --i-table galeeva_covid/dada2/galeeva_reverse_table.qza \
  --i-taxonomy taxonomic_analysis/galeeva_reverse_taxonomy.qza \
  --m-metadata-file  galeeva_covid/galeeva_metadata.tsv \
  --o-visualization taxonomic_analysis/galeeva_reverse_taxa-bar-plots.qzv

## Filter out mitochondria and chloroplast ##
qiime taxa filter-table \
  --i-table galeeva_covid/dada2/galeeva_reverse_table.qza \
  --i-taxonomy taxonomic_analysis/galeeva_reverse_taxonomy.qza \
  --p-exclude mitochondria,chloroplast \
  --o-filtered-table taxonomic_analysis/galeeva_reverse_table-no-mitochondria-no-chloroplast.qza

## Make into human readable file ##
qiime feature-table summarize \
  --i-table taxonomic_analysis/galeeva_reverse_table-no-mitochondria-no-chloroplast.qza \
  --o-visualization taxonomic_analysis/galeeva_reverse_table-no-mitochondria-no-chloroplast.qzv \
  --m-sample-metadata-file galeeva_covid/galeeva_metadata.tsv

## Make diversity_analysis folder in galeeva_covid ##
if [ ! -d diversity_analysis ]; then
    mkdir -p diversity_analysis
fi

## Generate a tree for phylogenetic diversity analyses ##
qiime phylogeny align-to-tree-mafft-fasttree \
  --i-sequences dada2/galeeva_reverse_rep-seqs.qza \
  --o-alignment diversity_analysis/galeeva_reverse_aligned-rep-seqs.qza \
  --o-masked-alignment diversity_analysis/galeeva_reverse_masked-aligned-rep-seqs.qza \
  --o-tree diversity_analysis/galeeva_reverse_unrooted-tree.qza \
  --o-rooted-tree diversity_analysis/galeeva_reverse_rooted-tree.qza

## Alpha-rarefaction ##
qiime diversity alpha-rarefaction \
  --i-table dada2/galeeva_reverse_table.qza \
  --i-phylogeny diversity_analysis/galeeva_reverse_rooted-tree.qza \
  --p-max-depth 40000 \
  --m-metadata-file galeeva_metadata.tsv \
  --o-visualization diversity_analysis/galeeva_reverse_alpha-rarefaction.qzv

## Make Galeeva export folder ##
mkdir galeeva_export

## Export table.qza ##
qiime tools export \
--input-path dada2/galeeva_reverse_table.qza \
--output-path galeeva_export/galeeva_table_export

## Convert feature biom to a feature table ##
biom convert \
-i feature-table.biom \
--to-tsv \
-o galeeva_feature-table.txt

## Export rooted_tree.qza ##
qiime tools export \
  --input-path diversity_analysis/galeeva_reverse_rooted-tree.qza \
  --output-path galeeva_export/galeeva_rooted-tree_export

## Export taxonomy.qza ##
qiime tools export \
  --input-path taxonomic_analysis/galeeva_reverse_taxonomy.qza \
  --output-path galeeva_covid/galeeva_export/galeeva_taxonomy_export
