## Trim primers using cutadapt ##
qiime cutadapt trim-paired \
            --i-demultiplexed-sequences almomani_demux_seqs.qza \
            --p-cores 16 \
            --p-front-f GTGCCAGCMGCCGCGGTAA \
            --p-front-r GACTACHVGGGTWTCTAAT \
            --p-discard-untrimmed \
            --p-no-indels \
            --o-trimmed-sequences almomani_reads_trimmed_pre.qza \
                        --verbose

## Adapter trimming ##
qiime cutadapt trim-paired \
            --i-demultiplexed-sequences almomani_demux_seqs.qza \
            --p-cores 16 \
                --p-adapter-r TTACCGCGGCNGCTGGCAC \
            --p-no-indels \
            --o-trimmed-sequences almomani_reads_trimmed.qza \
                        --verbose

## Make into human readable file ##
qiime demux summarize \
            --i-data almomani_reads_trimmed.qza \
            --o-visualization almomani_reads_trimmed_summary1.qzv

## DADA2 paired-end denoising ##
qiime dada2 denoise-paired \
    --i-demultiplexed-seqs almomani_reads_trimmed.qza \
    --p-trim-left-f 0 \
    --p-trim-left-r 0 \
    --p-trunc-len-f 232 \
    --p-trunc-len-r 217 \
    --p-n-threads 16 \
    --o-table dada2/table.qza \
    --o-representative-sequences dada2/rep-seqs.qza \
    --o-denoising-stats dada2/denoising-stats.qza \
    --verbose

## Summarize ASV table to visualize sequence counts per sample ##
qiime feature-table summarize \
    --i-table dada2/table.qza \
    --o-visualization dada2/table_summary.qzv

## Summarize representative sequences ##
qiime feature-table tabulate-seqs \
    --i-data dada2/rep-seqs.qza \
    --o-visualization dada2/rep-seqs_summary.qzv

## Visualize DADA2 denoising stats ##
qiime metadata tabulate \
    --m-input-file dada2/denoising-stats.qza \
    --o-visualization dada2/denoising-stats_summary.qzv

## Make taxonomic_analysis folder in the data folder ##
if [ ! -d taxonomic_analysis ]; then
    mkdir -p taxonomic_analysis
fi

## Extract amplicon of interest from the reference database ##
qiime feature-classifier extract-reads \
  --i-sequences /mnt/datasets/silva_ref_files/silva-138-99-seqs.qza \
  --p-f-primer GTGCCAGCMGCCGCGGTAA \
  --p-r-primer GACTACHVGGGTWTCTAAT \
  --o-reads taxonomic_analysis/ref-seqs-trimmed.qza

## Train classifier with new ref-seq file ##
qiime feature-classifier fit-classifier-naive-bayes \
  --i-reference-reads taxonomic_analysis/ref-seqs-trimmed.qza \
  --i-reference-taxonomy /mnt/datasets/silva_ref_files/silva-138-99-tax.qza \
  --o-classifier taxonomic_analysis/classifier.qza

## Use the trained classifier to assign taxonomy to your reads (rep-seqs.qza) ##
qiime feature-classifier classify-sklearn \
  --i-classifier taxonomic_analysis/classifier.qza \
  --i-reads almomani_covid/dada2/rep-seqs.qza \
  --o-classification taxonomic_analysis/almomani_taxonomy.qza

## Make into human readable file ##
qiime metadata tabulate \
  --m-input-file almomani_taxonomy.qza \
  --o-visualization almomani_taxonomy.qzv

## Taxonomy barplots ##
qiime taxa barplot \
  --i-table almomani_covid/dada2/table.qza \
  --i-taxonomy taxonomic_analysis/almomani_taxonomy.qza \
  --m-metadata-file  almomani_covid/almomani_metadata.tsv \
  --o-visualization taxonomic_analysis/almomani_taxa-bar-plots.qzv

## Filter out mitochondria and chloroplast ##
qiime taxa filter-table \
  --i-table almomani_covid/dada2/table.qza \
  --i-taxonomy taxonomic_analysis/almomani_taxonomy.qza \
  --p-exclude mitochondria,chloroplast \
  --o-filtered-table taxonomic_analysis/almomani_table-no-mitochondria-no-chloroplast.qza

## Make into human readable file ##
qiime feature-table summarize \
  --i-table taxonomic_analysis/almomani_table-no-mitochondria-no-chloroplast.qza \
  --o-visualization taxonomic_analysis/almomani_table-no-mitochondria-no-chloroplast.qzv \
  --m-sample-metadata-file almomani_covid/almomani_metadata.tsv

## Make diversity_analysis folder in almomani_covid ##
if [ ! -d diversity_analysis ]; then
    mkdir -p diversity_analysis
fi

## Generate a tree for phylogenetic diversity analyses ##
qiime phylogeny align-to-tree-mafft-fasttree \
  --i-sequences dada2/rep-seqs.qza \
  --o-alignment diversity_analysis/aligned-rep-seqs.qza \
  --o-masked-alignment diversity_analysis/masked-aligned-rep-seqs.qza \
  --o-tree diversity_analysis/unrooted-tree.qza \
  --o-rooted-tree diversity_analysis/rooted-tree.qza

## Alpha-rarefaction ##
qiime diversity alpha-rarefaction \
  --i-table dada2/table.qza \
  --i-phylogeny diversity_analysis/rooted-tree.qza \
  --p-max-depth 200000 \
  --m-metadata-file almomani_metadata.tsv \
  --o-visualization diversity_analysis/almomani_alpha-rarefaction.qzv

## Make almomani export folder ##
 mkdir almomani_export

## Export table.qza ##
qiime tools export \
--input-path dada2/table.qza \
--output-path almomani_export/almomani_table_export

## Convert feature biom to a feature table ##
biom convert \
-i feature-table.biom \
--to-tsv \
-o almomani_feature-table.txt

## Export rooted_tree.qza ##
qiime tools export \
  --input-path diversity_analysis/rooted-tree.qza \
  --output-path almomani_export/almomani_rooted-tree_export

## Export taxonomy.qza ##
qiime tools export \
  --input-path taxonomic_analysis/almomani_taxonomy.qza \
  --output-path almomani_covid/almomani_export/almomani_taxonomy_export
